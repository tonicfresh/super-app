From template/backend/src/module-registry.ts (register method):
```typescript
register(plugin: ModulePlugin): void {
  const existing = modules.find((m) => m.config.name === plugin.config.name);
  if (existing) {
    throw new Error(`Module "${plugin.config.name}" is already registered`);
  }
  modules.push({ ...plugin, registeredAt: new Date() });
}
```
