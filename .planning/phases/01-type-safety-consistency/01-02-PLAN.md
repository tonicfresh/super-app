    #!/usr/bin/env bun
    /**
     * Validierungsscript: Prueft ob Backend und Frontend identische Versionen
     * fuer gemeinsame Dependencies verwenden.
     * Aufruf: bun run scripts/check-version-sync.ts
     * Exit 0 = alles synchron, Exit 1 = Mismatches gefunden
     */

    import backendPkg from "../template/backend/package.json";
    import frontendPkg from "../template/frontend/package.json";

    const backendDeps: Record<string, string> = {
      ...((backendPkg as any).dependencies ?? {}),
      ...((backendPkg as any).devDependencies ?? {}),
    };
    const frontendDeps: Record<string, string> = {
      ...((frontendPkg as any).dependencies ?? {}),
      ...((frontendPkg as any).devDependencies ?? {}),
    };

    const mismatches: string[] = [];
    for (const [name, beVersion] of Object.entries(backendDeps)) {
      const feVersion = frontendDeps[name];
      if (feVersion && beVersion !== feVersion) {
        mismatches.push(`  ${name}: backend=${beVersion}  frontend=${feVersion}`);
      }
    }

    if (mismatches.length > 0) {
      console.error("Version mismatches found between backend and frontend:");
      mismatches.forEach((m) => console.error(m));
      process.exit(1);
    }

    console.log("All shared dependency versions are in sync.");
    process.exit(0);
    ```

    import { describe, it, expect } from "bun:test";
    import { readdir } from "fs/promises";
    import { join } from "path";

    /**
     * Schema-Prefix-Konventions-Test (CON-02).
     * Prueft dass alle Drizzle Schema-Dateien die richtige Prefix-Konvention verwenden:
     * - Framework: base_*
     * - App (template/backend/src/db/): app_*
     * - Module: <moduleName>_*
     * - Ausnahme: template/backend/src/ai/db/schema.ts darf mc_* verwenden (D-11)
     */

    const PROJECT_ROOT = join(import.meta.dir, "../../../..");

    // Erlaubte Prefixes pro Verzeichnis
    const ALLOWED_PREFIXES: Record<string, string[]> = {
      "template/backend/src/db": ["app_"],
      "template/backend/src/ai/db": ["mc_"], // Per D-11: MC-Domain
      "modules/mission-control/backend/src/db": ["mc_"],
      "modules/todos/backend/src/db": ["todos_"],
    };

    async function readSchemaFiles(dir: string): Promise<string[]> {
      try {
        const files = await readdir(join(PROJECT_ROOT, dir));
        return files.filter((f) => f.endsWith("schema.ts") || f === "schema.ts");
      } catch {
        return [];
      }
    }

    async function extractPrefixes(filePath: string): Promise<string[]> {
      const content = await Bun.file(join(PROJECT_ROOT, filePath)).text();
      const matches = content.match(/pgTableCreator\(\(name\)\s*=>\s*`(\w+?)_\$\{name\}`\)/g);
      if (!matches) return [];
      return matches.map((m) => {
        const prefixMatch = m.match(/`(\w+?)_\$/);
        return prefixMatch ? `${prefixMatch[1]}_` : "";
      }).filter(Boolean);
    }

    describe("Schema Prefix Convention (CON-02)", () => {
      for (const [dir, allowed] of Object.entries(ALLOWED_PREFIXES)) {
        it(`${dir} should only use prefix(es): ${allowed.join(", ")}`, async () => {
          const schemaFiles = await readSchemaFiles(dir);
          for (const file of schemaFiles) {
            const filePath = `${dir}/${file}`;
            const prefixes = await extractPrefixes(filePath);
            for (const prefix of prefixes) {
              expect(allowed).toContain(prefix);
            }
          }
        });
      }

      it("should not have any schema files with push_ prefix", async () => {
        const schemaFiles = await readSchemaFiles("template/backend/src/db");
        for (const file of schemaFiles) {
          const content = await Bun.file(
            join(PROJECT_ROOT, `template/backend/src/db/${file}`)
          ).text();
          expect(content).not.toContain("`push_${name}`");
        }
      });
    });
    ```
