    **Permission-Middleware** (`template/backend/src/auth/permission-middleware.ts`) — per D-01, D-02, D-03:

    ```typescript
    import { hasPermission } from "@framework/lib/auth/permissions";
    import type { Context, Next } from "hono";

    /**
     * Super-App Permission-Middleware.
     * Ersetzt die deaktivierte checkUserPermission aus dem Framework (HACK).
     * Nutzt Framework hasPermission() mit bestehendem Cache (per D-02, D-03).
     */
    export async function checkPermission(c: Context, next: Next) {
      const userId = c.get("usersId");
      if (!userId) {
        return c.json({ error: "Unauthorized" }, 401);
      }

      const method = c.req.method;
      const path = c.req.path;
      const allowed = await hasPermission(userId, method, path);

      if (!allowed) {
        return c.json({ error: "Forbidden" }, 403);
      }

      await next();
    }
    ```

    **Auth-Error-Handler** (`template/backend/src/auth/auth-error-handler.ts`) — per D-04, D-05:

    ```typescript
    import type { Context, Next } from "hono";

    /**
     * Super-App Auth-Wrapper mit granularer Fehlerbehandlung.
     * Wraps Framework authAndSetUsersInfo und differenziert Fehlertypen.
     */
    export function createAuthErrorHandler(
      authFn: (c: Context, next: Next) => Promise<void | Response>
    ) {
      return async (c: Context, next: Next) => {
        try {
          await authFn(c, next);
        } catch (error) {
          const message = error instanceof Error ? error.message : "Unknown error";

          // Hanko Service nicht erreichbar
          if (message.includes("fetch failed") || message.includes("ECONNREFUSED") || message.includes("ETIMEDOUT")) {
            return c.json({ error: "Auth service unavailable" }, 503);
          }

          // Token expired/invalid — generischer 401
          return c.json({ error: "Unauthorized" }, 401);
        }
      };
    }
    ```

    WICHTIG: `createAuthErrorHandler` ist ein Wrapper-Factory (per D-05), nicht eine eigenstaendige Auth-Funktion. Sie nimmt `authAndSetUsersInfo` als Parameter und fuegt Error-Differenzierung hinzu. Das vermeidet Duplizierung der Auth-Logik.
