    ```typescript
    // Cache-Wrapper fuer Cost Queries (per D-09: 5min daily, 1min module)
    const dailyTotalCache = { value: 0, expiresAt: 0 };
    const moduleDailyCache = new Map<string, { value: number; expiresAt: number }>();

    const cachedQueryDailyTotal = async (): Promise<number> => {
      if (dailyTotalCache.expiresAt > Date.now()) return dailyTotalCache.value;
      const value = await getCostQueries().queryDailyTotal();
      dailyTotalCache.value = value;
      dailyTotalCache.expiresAt = Date.now() + 5 * 60 * 1000; // 5 Minuten TTL
      return value;
    };

    const cachedQueryModuleDaily = async (moduleName: string): Promise<number> => {
      const cached = moduleDailyCache.get(moduleName);
      if (cached && cached.expiresAt > Date.now()) return cached.value;
      const value = await getCostQueries().queryModuleDaily(moduleName);
      moduleDailyCache.set(moduleName, { value, expiresAt: Date.now() + 60 * 1000 }); // 1 Minute TTL
      return value;
    };
    ```

    ```typescript
    function estimateCostUsd(
      model: string,
      tokensInput: number,
      tokensOutput: number,
      pricing: PricingTable
    ): number {
      const modelLower = model.toLowerCase();
      let pricePerMillion = { input: 3.0, output: 15.0 }; // Default: Claude Sonnet

      for (const [pattern, price] of Object.entries(pricing)) {
        if (modelLower.includes(pattern)) {
          pricePerMillion = price;
          break;
        }
      }

      return (
        (tokensInput / 1_000_000) * pricePerMillion.input +
        (tokensOutput / 1_000_000) * pricePerMillion.output
      );
    }
    ```
